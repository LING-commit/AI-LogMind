## 1. 插件系统抽象接口

### 1.1 接口层次结构

```
IPlugin (基类)
  ├── ICollector  (采集器) — parse / can_parse / watch_patterns
  ├── IAnalyzer   (分析器) — analyze
  └── IAlert      (告警器) — send
```

### 1.2 头文件 `include/logmind/plugin.h`

```cpp
#pragma once
#include <logmind/log_entry.h>
#include <string>
#include <vector>

namespace logmind {

// ──────────────────────────────────────────────────
// 插件类型元信息，供 PluginLoader 和 D-Bus 使用
// ──────────────────────────────────────────────────
struct PluginManifest {
    std::string name;           // 唯一标识符
    std::string version;        // SemVer
    std::string description;
    std::string author;
    PluginType  type;
    std::string license;
    bool        has_side_effects; // 是否涉及外部 I/O
};

// ──────────────────────────────────────────────────
// 生命周期回调（可选实现，默认空）
// ──────────────────────────────────────────────────
class IPluginLifecycle {
public:
    virtual ~IPluginLifecycle() = default;

    // 插件被加载后调用，用于初始化资源
    virtual bool on_load(const nlohmann::json& config) { return true; }

    // 插件被卸载前调用，用于释放资源
    virtual void on_unload() {}

    // 配置热更新时调用
    virtual bool on_reload_config(const nlohmann::json& new_config) { return true; }
};

// ──────────────────────────────────────────────────
// 基础插件接口
// ──────────────────────────────────────────────────
class IPlugin : public IPluginLifecycle {
public:
    ~IPlugin() override = default;

    // ── 元信息 ──
    virtual PluginManifest manifest() const = 0;

    // ── 帮助函数 ──
    const std::string& name()        const { static std::string s; return s = manifest().name; }
    const std::string& version()     const { static std::string s; return s = manifest().version; }
    const std::string& description() const { static std::string s; return s = manifest().description; }
    PluginType          type()       const { return manifest().type; }
};

} // namespace logmind
```

### 1.3 头文件 `include/logmind/collector.h`

```cpp
#pragma once
#include <logmind/plugin.h>
#include <istream>
#include <string_view>

namespace logmind {

// ──────────────────────────────────────────────────
// 采集器插件
// ──────────────────────────────────────────────────
class ICollector : public IPlugin {
public:
    PluginManifest manifest() const override = 0;

    // 判断是否能够解析该文件（基于文件名判断）
    // 返回值 > 0: 置信度（100=完全匹配），0: 不能处理
    virtual int match(const std::string& filename) const = 0;

    // 从输入流解析日志
    // 返回 LogEntry 列表 + 解析统计
    struct ParseResult {
        std::vector<LogEntry> entries;
        size_t lines_read     = 0;
        size_t lines_parsed   = 0;
        size_t lines_skipped  = 0;  // 不符合格式的行数
        std::string error;          // 非空表示存在解析错误
    };

    virtual ParseResult parse(std::istream& stream) = 0;

    // ── 可选覆写 ──

    // 批处理模式：一次性解析整个文件（默认按行调用 parse）
    virtual ParseResult parse_file(const std::string& path);

    // 声明该插件关心的文件 glob 模式
    virtual std::vector<std::string> watch_patterns() const { return {}; }

    // 是否支持增量解析（日志轮转时使用，默认 false）
    virtual bool supports_incremental() const { return false; }
};

} // namespace logmind
```

### 1.4 头文件 `include/logmind/analyzer.h`

```cpp
#pragma once
#include <logmind/plugin.h>
#include <functional>

namespace logmind {

// ──────────────────────────────────────────────────
// 分析器插件
// ──────────────────────────────────────────────────
class IAnalyzer : public IPlugin {
public:
    PluginManifest manifest() const override = 0;

    // 对一批日志执行分析
    // 可修改 LogEntry::tags / 过滤掉不关心的条目
    virtual std::vector<LogEntry> analyze(std::vector<LogEntry> entries) = 0;

    // ── AI 分析器扩展接口（可选实现） ──
    struct AIAnalysis {
        std::string log_id;
        std::string summary;               // 根因摘要
        std::vector<std::string> root_causes; // 可能根因列表
        std::vector<std::string> suggestions; // 修复建议
        double confidence;                 // 0.0 ~ 1.0
        std::string raw_response;          // LLM 原始响应
    };

    // 是否支持 AI 分析
    virtual bool supports_ai_analysis() const { return false; }

    // 对单条日志做深度 AI 分析
    virtual AIAnalysis ai_analyze(const LogEntry& entry) {
        (void)entry;
        return {};
    }

    // ── 流式分析接口（高阶，可选实现） ──
    using AnalyzeCallback = std::function<void(std::vector<LogEntry>)>;

    // 流模式：每到达一批日志就回调（默认退化为 batch analyze）
    virtual void analyze_stream(
        std::vector<LogEntry> entries,
        AnalyzeCallback callback)
    {
        auto result = analyze(std::move(entries));
        if (!result.empty()) {
            callback(std::move(result));
        }
    }
};

} // namespace logmind
```

### 2.2 模块职责

| 模块 | 职责 | 关键类 |
|------|------|--------|
| **DaemonCore** | 状态机管理、组件组装、生命周期协调 | `DaemonCore` |
| **ConfigMgr** | JSON 配置文件读写、热重载、参数校验 | `ConfigMgr` |
| **PluginLoader** | .so 扫描、dlopen/dlsym、插件实例管理、启用/禁用 | `PluginLoader` |
| **FileWatcher** | inotify 事件监听、增量读取、日志轮转感知 | `FileWatcher` |
| **Pipeline** | Collector→Analyzer→Alert 管线编排、线程调度 | `Pipeline` |
| **LogBuffer** | 环形缓冲区，存储最近 N 条日志，支持 D-Bus 查询 | `LogBuffer` |
| **VectorStore** | SQLite + HNSW 向量索引，embedding 存储与检索 | `VectorStore` |
| **DaemonAdaptor** | D-Bus 服务注册、方法分发、信号推送 | `DaemonAdaptor` |

### 2.4 状态定义

```cpp
enum class DaemonState : uint8_t {
    INIT,       // 初始状态
    LOADING,    // 加载插件
    STARTING,   // 启动服务
    RUNNING,    // 正常运行
    RELOADING,  // 重载插件/配置
    STOPPING,   // 正在停止
    STOPPED,    // 已停止
    FAILED      // 启动失败
};

const char* to_string(DaemonState s) {
    switch (s) {
        case DaemonState::INIT:      return "INIT";
        case DaemonState::LOADING:   return "LOADING";
        case DaemonState::STARTING:  return "STARTING";
        case DaemonState::RUNNING:   return "RUNNING";
        case DaemonState::RELOADING: return "RELOADING";
        case DaemonState::STOPPING:  return "STOPPING";
        case DaemonState::STOPPED:   return "STOPPED";
        case DaemonState::FAILED:    return "FAILED";
    }
    return "UNKNOWN";
}
```

### 2.6 线程模型

```
┌─────────────────────────────────────────────────────────┐
│                     主线程 (Qt Event Loop)                │
│  DaemonCore / D-Bus Adaptor / Qt Signals/Slots           │
│  负责：信号分发、D-Bus 查询、状态机管理                    │
└─────────────────────────┬───────────────────────────────┘
                          │
    ┌─────────────────────┼─────────────────────┐
    │                     │                     │
    ▼                     ▼                     ▼
┌────────────┐  ┌──────────────────┐  ┌──────────────────┐
│ FileWatcher│  │ Pipeline Worker   │  │ AI Analyzer Pool  │
│ Thread     │  │ Thread Pool       │  │ (std::async)      │
│ (单线程)    │  │ (可配置,默认4)    │  │ (最多2并发)       │
│            │  │                   │  │                   │
│ inotify    │  │ parse + analyze   │  │ LLM HTTP 调用     │
│ read+seek  │  │ + alert check     │  │ (阻塞 I/O，隔离)  │
└────────────┘  └──────────────────┘  └──────────────────┘
```

**线程安全约定**：
- `PluginLoader` 的所有方法在主线程调用，D-Bus 自动序列化
- `Pipeline::process_raw` 由 FileWatcher 线程调用，内部加锁写 LogBuffer
- `IAnalyzer::analyze` 可能被多线程并发调用，插件必须线程安全
- `IAlert::send` 在 Pipeline Worker 线程调用，告警器应自行异步发送

```cpp
// LogEntry ↔ QVariantMap 序列化工具函数
namespace {

QVariantMap log_entry_to_varmap(const logmind::LogEntry& entry) {
    QVariantMap map;
    map["source"]    = QString::fromStdString(entry.source);
    map["raw"]       = QString::fromStdString(entry.raw);
    map["timestamp"] = QVariant::fromValue(
        std::chrono::system_clock::to_time_t(entry.timestamp));
    map["level"]     = QString::fromStdString(logmind::to_string(entry.level));
    map["message"]   = QString::fromStdString(entry.message);
    map["module"]    = QString::fromStdString(entry.module);
    map["fields"]    = QString::fromStdString(entry.fields.dump());
    map["tags"]      = QString::fromStdString(entry.tags.dump());
    map["meta"]      = QString::fromStdString(entry.meta.dump());
    return map;
}
```

### 3.4 错误码定义

```cpp
// D-Bus 方法返回的错误码约定
// 成功时 success=true, error=""；失败时 success=false, error=<错误码>
enum class DBusError : uint8_t {
    OK                   = 0,
    PLUGIN_NOT_FOUND     = 1,  // 找不到指定插件
    PLUGIN_DISABLED      = 2,  // 插件已禁用
```

### 3.5 权限校验实现

```cpp
bool DaemonAdaptor::check_permission(
    const QDBusMessage& msg,
    const QSet<QString>& allowed_methods)
{
    // 普通查询方法允许所有用户
    static const QSet<QString> public_methods = {
        "QueryLogs", "QueryLogsByTime",
        "ListPlugins", "GetPluginInfo",
        "GetStats", "GetStatus"
    };

    // 获取调用的方法名
    QString method = msg.member();

    if (public_methods.contains(method)) {
        return true;  // 公开接口无需校验
    }

    // 管理接口需要检查用户组
    uid_t uid = msg.uid();
    // 检查 uid 是否属于 logmind-adm 或 logmind-dev 组
    return is_member_of_group(uid, "logmind-adm") ||
           is_member_of_group(uid, "logmind-dev");
}
```

### 6.3 LogModel 核心实现要点

```cpp
// src/client/models/log_model.cpp

void LogModel::appendLog(const LogItem& item) {
    // 从 D-Bus 子线程调用 → 需要跨线程安全
    // Qt 的 signal/slot 自动将 appendLog 调用排队到 UI 线程

    beginInsertRows({}, m_items.size(), m_items.size());
    m_items.push_back(item);
    endInsertRows();

    // 更新时间范围
    if (m_items.size() == 1) {
        m_earliest_ts = item.timestamp_ns;
    }
    m_latest_ts = item.timestamp_ns;
    m_count_total++;

    if (item.level == "ERROR" || item.level == "FATAL") {
        m_error_count++;
        emit errorCountChanged(m_error_count);
    }

    enforce_capacity();
    emit logCountChanged(static_cast<int>(m_items.size()), m_count_total);
}

void LogModel::enforce_capacity() {
    if (m_items.size() <= static_cast<size_t>(m_capacity)) return;

    int over = static_cast<int>(m_items.size()) - m_capacity;
    beginRemoveRows({}, 0, over - 1);
    for (int i = 0; i < over; ++i) {
        m_items.pop_front();
    }
    endRemoveRows();

    // 更新最早时间戳
    if (!m_items.empty()) {
        m_earliest_ts = m_items.front().timestamp_ns;
    }
    emit bufferTrimmed(over);
}

QVariant LogModel::data(const QModelIndex& index, int role) const {
    if (!index.isValid() || index.row() >= static_cast<int>(m_items.size()))
        return {};

    const auto& item = m_items[index.row()];

    switch (role) {
    case IdRole:            return item.id;
    case SourceRole:        return item.source;
    case RawRole:           return item.raw;
    case TimestampRole:     return QDateTime::fromMSecsSinceEpoch(item.timestamp_ns / 1000000);
    case LevelRole:         return item.level;
    case MessageRole:       return item.message;
    case ModuleRole:        return item.module;
    case FieldsRole:        return item.fields;
    case TagsRole:          return item.tags;
    case MetaRole:          return item.meta;
    case HasAiReportRole:   return item.has_ai_report;
    case AiSummaryRole:     return item.ai_summary;
    case LevelColorRole:    return level_color(item.level);
    case IsErrorRole:       return item.level == "ERROR" || item.level == "FATAL";
    default:                return {};
    }
}
```

```cpp
    QString m_level;    // 空 = 不过滤级别
    QString m_source;   // 空 = 不过滤来源
    qint64  m_time_start = 0;
    qint64  m_time_end   = 0;
};

// ── filterAcceptsRow 实现逻辑 ──
bool LogFilterModel::filterAcceptsRow(int row, const QModelIndex& parent) const {
    auto src = sourceModel();
    auto idx = src->index(row, 0, parent);

    auto level   = src->data(idx, LogModelRole::LevelRole).toString();
    auto source  = src->data(idx, LogModelRole::SourceRole).toString();
    auto message = src->data(idx, LogModelRole::MessageRole).toString();
    auto ts      = src->data(idx, LogModelRole::TimestampRole).toDateTime().toMSecsSinceEpoch();

    // 级别过滤
    if (!m_level.isEmpty() && level != m_level) return false;

    // 来源过滤
    if (!m_source.isEmpty() && source != m_source) return false;

    // 时间范围过滤
    if (m_time_start > 0 && ts < m_time_start) return false;
    if (m_time_end   > 0 && ts > m_time_end)   return false;

    // 文本搜索（多字段模糊匹配）
    if (!m_text.isEmpty()) {
        bool matched = message.contains(m_text, Qt::CaseInsensitive)
                    || source.contains(m_text, Qt::CaseInsensitive)
                    || level.contains(m_text, Qt::CaseInsensitive);
        if (!matched) return false;
    }

    return true;
}
```

### 7.2 HNSW 索引结构

```cpp
// src/daemon/vector_store_impl.h (内部实现)
#pragma once
#include <vector>
#include <queue>
#include <random>
#include <cmath>
#include <limits>
#include <unordered_map>

namespace logmind {

class HNSWIndex {
public:
    HNSWIndex(size_t max_elements,
              size_t M = 16,
              size_t ef_construction = 200);

    // ── HNSW 核心参数 ──
    size_t M()               const { return m_M; }
    size_t M_max()           const { return m_M; }
    size_t M_max0()          const { return 2 * m_M; }
    size_t ef_construction() const { return m_ef_construction; }
    size_t ef_search()       const { return m_ef_search; }
    void set_ef_search(size_t ef) { m_ef_search = ef; }

    // ── 索引操作 ──
    void add_point(const std::vector<float>& vec, size_t id);
    std::vector<size_t> search_knn(const std::vector<float>& query, size_t k) const;

    size_t size() const { return m_size; }
    bool   empty() const { return m_size == 0; }

private:
    // ── HNSW 层级结构 ──
    struct Node {
        size_t id;
        std::vector<std::pair<size_t, float>> neighbors;  // (neighbor_id, distance)
        int level;
    };

    // ── 随机层级生成（指数衰减） ──
    int random_level();

    // ── 距离计算 (L2) ──
    static float l2_distance(const std::vector<float>& a, const std::vector<float>& b);

    // ── 层级选择贪婪搜索 ──
    struct Candidate {
        size_t id;
        float  dist;
        bool operator>(const Candidate& other) const { return dist > other.dist; }
    };

    // 从 level 层开始的逐层搜索
    std::priority_queue<Candidate> search_layer(
        const std::vector<float>& query,
        size_t entry_point,
        size_t ef,
        int layer) const;

    // ── 成员变量 ──
    const size_t m_M;                // 每层最大连接数
    const size_t m_M_max;            // 非 0 层最大连接
    const size_t m_M_max0;           // 0 层 (底层) 最大连接 = 2*M
    const size_t m_ef_construction;  // 构建时搜索宽度
    size_t       m_ef_search = 50;   // 检索时搜索宽度

    size_t m_size = 0;               // 当前节点数
    int    m_max_level = 0;          // 当前最高层级
    size_t m_entry_point = 0;        // 入口节点 ID

    // 存储: level_id → { id → list of (neighbor_id, distance) }
    // 底层 (level 0) 包含所有节点
    std::unordered_map<int, std::unordered_map<size_t, Node>> m_graph;

    // 向量数据: id → vector
    std::vector<std::vector<float>> m_vectors;

    mutable std::mt19937 m_rng;
};

// ── 距离计算 ──
float HNSWIndex::l2_distance(const std::vector<float>& a, const std::vector<float>& b) {
    float sum = 0.0f;
    for (size_t i = 0; i < a.size(); ++i) {
        float d = a[i] - b[i];
        sum += d * d;
    }
    return std::sqrt(sum);
}
```

```cpp
    if (rc != SQLITE_OK) return false;
    m_impl->init_schema();

    // 2. 加载已有计数
    sqlite3_stmt* stmt;
    sqlite3_prepare_v2(m_impl->db, "SELECT COUNT(*) FROM vectors", -1, &stmt, nullptr);

        m_impl->config.max_elements,
        m_impl->config.hnsw_m,
        m_impl->config.hnsw_ef_construction

    if (m_impl->count > 0) {
```

### 8.4 错误处理策略

```cpp
AIAnalysis AiAnalyzer::ai_analyze(const LogEntry& entry) {
    AIAnalysis result;
    result.log_id = entry.meta.value("log_id", "unknown");

    for (int attempt = 0; attempt <= m_config.max_retries; ++attempt) {
        try {
            // ── 超时控制 ──
            auto future = std::async(std::launch::async, [this, &entry]() {
                auto prompt = build_prompt(entry);
                return call_llm(prompt);
            });

            auto status = future.wait_for(
                std::chrono::seconds(m_config.timeout_seconds));

            if (status == std::future_status::timeout) {
                // 超时：记录并重试或降级
                LOG_WARN("AI analysis timed out after {}s (attempt {}/{})",
                         m_config.timeout_seconds, attempt, m_config.max_retries);
                continue;  // 重试
            }

            auto raw = future.get();
            result = parse_response(raw);
            result.log_id = entry.meta.value("log_id", "unknown");
            result.raw_response = raw;
            return result;  // 成功

        } catch (const std::exception& e) {
            LOG_ERROR("AI analysis failed (attempt {}/{}): {}",
                      attempt, m_config.max_retries, e.what());

            // 判断错误类型
            if (is_network_error(e)) {
                // 网络错误：短暂等待后重试
                std::this_thread::sleep_for(std::chrono::seconds(1 << attempt));
                continue;
            }

            if (is_auth_error(e)) {
                // 认证错误（API Key 无效）：不重试，直接降级
                LOG_ERROR("AI authentication failed, skipping retry");
                break;
            }

            if (is_rate_limit_error(e)) {
                // 限流：等待后重试
                std::this_thread::sleep_for(std::chrono::seconds(5));
                continue;
            }
        }
    }

    // ── 所有重试耗尽：降级 ──
    result.summary = "[AI 分析不可用] 请在 daemon 日志中检查 AI 配置和网络连接";
    result.confidence = 0.0;
    result.root_causes = {"AI 服务暂不可用，请稍后重试或检查配置"};
    result.suggestions = {
        "检查 AI API Key 是否有效",
        "确认 Ollama/OpenAI 服务是否运行",
        "检查网络连通性和防火墙规则"
    };
    return result;
}

// ── 错误辅助判断 ──
bool AiAnalyzer::is_network_error(const std::exception& e) {
    // 匹配 curl 错误码: CURLE_COULDNT_CONNECT, CURLE_OPERATION_TIMEDOUT 等
    std::string msg = e.what();
    return msg.find("Couldn't connect") != std::string::npos
        || msg.find("timeout") != std::string::npos
        || msg.find("resolve host") != std::string::npos;
}

bool AiAnalyzer::is_auth_error(const std::exception& e) {
    return std::string(e.what()).find("401") != std::string::npos
        || std::string(e.what()).find("403") != std::string::npos;
}

bool AiAnalyzer::is_rate_limit_error(const std::exception& e) {
    return std::string(e.what()).find("429") != std::string::npos;
}
```

### 8.5 重试与降级策略总结

| 错误类型 | 处理方式 | 重试次数 | 等待策略 | 降级结果 |
|----------|----------|----------|----------|----------|
| 超时 | 重试 | 最多 2 次 | 立即重试 | "分析超时"提示 |
| 网络不可达 | 重试 | 最多 2 次 | 指数退避 (1s, 2s) | "服务不可用"降级 |
| HTTP 401/403 | 不重试 | 0 | — | "API Key 无效"提示 |
| HTTP 429 限流 | 重试 | 最多 2 次 | 等待 5s | "服务繁忙"降级 |
| HTTP 5xx | 重试 | 最多 2 次 | 指数退避 | "服务错误"降级 |
| 响应解析失败 | 不重试 | 0 | — | "响应格式异常"降级 |

### 9.5 清理策略

#### 9.5.1 内存清理（LogBuffer + Qt LogModel）

| 策略 | 触发条件 | 行为 |
|------|----------|------|
| **容量淘汰** | LogBuffer 超过容量上限 | FIFO 淘汰最旧 10% 的条目，记录 discarded 计数 |
| **客户端限容** | LogModel 超过容量 | 同理 FIFO 淘汰，发射 `bufferTrimmed` 信号 |
| **时间窗口** | （可配置）仅保留最近 N 小时 | 兼容容量淘汰，如果 N=0 则禁用 |

#### 9.5.2 SQLite 持久化清理

| 策略 | 触发条件 | 行为 |
|------|----------|------|
| **按天分区淘汰** | 定时清理任务（默认每 6 小时） | `DROP TABLE IF EXISTS logs_<30天前>` |
| **空间阈值淘汰** | 数据库文件 > max_db_size_mb | 从最旧分区开始删除，直到低于阈值 80% |
| **VACUUM** | 每次清理后空闲页 > 20% | `PRAGMA auto_vacuum=INCREMENTAL;` 回收磁盘空间 |

### 9.6 磁盘空间保护

```cpp
// src/daemon/disk_monitor.h
#pragma once
#include <QObject>
#include <QTimer>
#include <filesystem>

class DiskMonitor : public QObject {
    Q_OBJECT
public:
    DiskMonitor(const std::string& watch_path, QObject* parent = nullptr);

    void start();
    void stop();

    // 当前磁盘状态
    float free_gb()    const { return m_free_gb; }
    float usage_pct()  const { return m_usage_pct; }
    bool  is_critical() const { return m_usage_pct > 95.0f; }

signals:
    void warningLowDisk(float free_gb, float usage_pct);
    void criticalLowDisk(float free_gb, float usage_pct);
    void diskRecovered(float free_gb, float usage_pct);

private slots:
    void check_disk();

private:
    std::filesystem::path m_path;
    QTimer* m_timer = nullptr;

    float m_free_gb   = 0;
    float m_usage_pct = 0;
    bool  m_was_critical = false;
    bool  m_was_warning  = false;
};

void DiskMonitor::check_disk() {
    auto space_info = std::filesystem::space(m_path);
    m_free_gb   = static_cast<float>(space_info.free) / (1024 * 1024 * 1024);
    m_usage_pct = 100.0f - (static_cast<float>(space_info.free) / space_info.capacity * 100.0f);

    if (m_usage_pct > 95.0f) {
        if (!m_was_critical) {
            emit criticalLowDisk(m_free_gb, m_usage_pct);
            m_was_critical = true;
        }
    } else if (m_usage_pct > 85.0f) {
        if (!m_was_warning) {
            emit warningLowDisk(m_free_gb, m_usage_pct);
            m_was_warning = true;
        }
        m_was_critical = false;
    } else {
        if (m_was_critical || m_was_warning) {
            emit diskRecovered(m_free_gb, m_usage_pct);
        }
        m_was_critical = false;
        m_was_warning  = false;
    }
}
```
